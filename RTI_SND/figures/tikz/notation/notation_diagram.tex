%% ── CONTINUOUS (left half) ──────────────────────────────────────

\node[root]  (root)  at (0,0)
  {$\phi_{cijm}$\\[2pt]{\footnotesize root form $(i,j,c,m)$}};

\node[cont1] (Fcij)  at (-3.6, 3.4)
  {$\phi_{cij}=\displaystyle\sum_{m}\phi_{cijm}$\\[2pt]
   {\footnotesize $m$ suppressed}};

\node[cont1] (Fijm)  at ( 3.6, 3.4)
  {$\phi_{ijm}=\displaystyle\sum_{c}\phi_{cijm}$\\[2pt]
   {\footnotesize $c$ suppressed}};

\node[cont2] (Fij)   at (0, 6.8)
  {$\phi_{ij}=\displaystyle\sum_{c}\sum_{m}\phi_{cijm}$\\[2pt]

\draw[arr] (root) -- (Fcij)
  node[lbl,midway,left=3pt]{$\textstyle\sum_{m\in\mathcal{M}}$};

  node[lbl,midway,right=3pt]{$\textstyle\sum_{c\in C}$};

  node[lbl,midway,right=3pt]{$\textstyle\sum_{m\in\mathcal{M}}$};

%% ── DOT NOTATION (below root) ───────────────────────────────────

\node[cont1, dashed] (dotj) at (-3.6,-3.4)

\node[cont1, dashed] (doti) at ( 3.6,-3.4)
  {$\phi_{ci\cdot m}=\displaystyle\sum_{j\in V}\phi_{cijm}$\\[2pt]

  node[lbl,midway,left=3pt]{$\textstyle\sum_{i\in V}$};
\draw[arr] (root) -- (doti)
  node[lbl,midway,right=3pt]{$\textstyle\sum_{j\in V}$};

%% ── BINARY (right half) ─────────────────────────────────────────

\node[root] (Xroot) at (11, 0)
  {$X_{cijm}$\\[2pt]{\footnotesize root binary $(i,j,c,m)$}};

\node[binOR] (XORc) at (8, 3.4)
  {$\overline{X}_{cij}$\;(new variable)\\[2pt]
   {\footnotesize $m$ suppressed}\\[1pt]
   {\footnotesize \eqref{eq:or-m-lb}: $X_{cijm}\leq\overline{X}_{cij}$}\\
   {\footnotesize \eqref{eq:or-m-ub}: $\overline{X}_{cij}\leq\sum_m X_{cijm}$}};

\node[binOR] (XORm) at (14, 3.4)
  {$\overline{X}_{ijm}$\;(new variable)\\[2pt]
   {\footnotesize $c$ suppressed}\\[1pt]
   {\footnotesize \eqref{eq:or-c-lb}: $X_{cijm}\leq\overline{X}_{ijm}$}\\
   {\footnotesize \eqref{eq:or-c-ub}: $\overline{X}_{ijm}\leq\sum_c X_{cijm}$}};

\node[binHAT] (XHAT) at (11,-3.4)
  {$\hat{X}_{cij}$\;(new variable)\\[2pt]
   {\footnotesize $m$ suppressed}\\[1pt]
   {\footnotesize \eqref{eq:exactly-one}: $\sum_{c\in C_{ij}}\hat{X}_{cij}=1$}};

\draw[barr] (Xroot) -- (XORc)
  node[blbl,midway,left=3pt]{OR over $m$};
\draw[barr] (Xroot) -- (XORm)
  node[blbl,midway,right=3pt]{OR over $c$};
\draw[barr] (Xroot) -- (XHAT)
  node[blbl,midway,right=3pt]{exactly-one over $c$};
