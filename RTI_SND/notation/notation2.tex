Variables in the model carry up to five indices (see
Table~\ref{tab:nomenclature}). To avoid writing three and four-level
summations and writing additional constraints to encode Boolean logic,
we adopt the following convention: suppressing one or more indices from
a variable and placing an accent on its name produces a derived
variable whose values are obtained by collapsing the suppressed indices
according to one of three aggregation rules: summation
($\overline{\phantom{x}}$), logical OR
($\widecheck{\phantom{x}}$), or at-most-one
($\widehat{\phantom{x}}$). The accent identifies which rule is
applied.

\begin{convention}[Index suppression and aggregation]
\label{conv:notation}

We call the original, fully-indexed variable the \emph{root variable}.
Each index of a root variable ranges over a set. The sets over which
the retained indices range are called the \emph{projected sets}, and
their Cartesian product is denoted~$\Pi$; the sets over which the
suppressed indices range are called the \emph{suppressed sets}, and
their Cartesian product is denoted~$\mathcal{S}$. We write
$\pi\in\Pi$ for a generic element of the projected domain and
$\sigma\in\mathcal{S}$ for a generic element of the suppressed domain.
A derived variable is defined for every $\pi\in\Pi$ by collapsing all
$\sigma\in\mathcal{S}$ according to the aggregation rule indicated by
the accent.

Whenever a derived variable appears in any constraint of the model,
the variable and its defining constraints (as specified by the
applicable rule below) are automatically introduced into the
formulation for every $\pi\in\Pi$. In other words, writing a derived
variable in a constraint is a shorthand that simultaneously declares
the variable and appends its linking constraints to the model; no
separate definition is needed.

%----------------------------------------------------------------------
\medskip
\noindent\textbf{Rule~1 \textnormal{(Sum,~$\overline{\phantom{x}}$).}}
Suppressing one or more indices and writing a
$\overline{\phantom{x}}$ over the root variable denotes the sum over
the suppressed domain~$\mathcal{S}$. For every $\pi\in\Pi$ the
derived variable is defined by:
\[
  \overline{\phi}_{\pi}
  \;\coloneqq\;
  \sum_{\sigma\in\mathcal{S}} \phi_{\pi,\sigma}
  \quad \forall\,\pi\in\Pi.
\]
For continuous and integer root variables the result is standard
summation; for binary root variables it represents an integer count of
active entries, not a Boolean sum.

\smallskip\noindent\textit{Example.}
From the root variable $NF_{ijmpr}$
$\bigl((i,j)\!\in\!E,\;m\!\in\!\mathcal{M},\;p\!\in\!P,
  \;r\!\in\!R\bigr)$, suppressing $p$
($\Pi=E\times\mathcal{M}\times R$, $\mathcal{S}=P$) and
suppressing all indices
($\Pi=\emptyset$, $\mathcal{S}=E\times\mathcal{M}\times P\times R$):
\begin{align}
  \overline{NF}_{ijmr}
  &= \sum_{p\in P} NF_{ijmpr}
  && \forall\,(i,j)\in E,\;m\in\mathcal{M},\;r\in R,
  \label{eq:sum-ex1} \\[4pt]
  \overline{NF}
  &= \sum_{(i,j)\in E}\sum_{m\in\mathcal{M}}\sum_{p\in P}\sum_{r\in R} NF_{ijmpr}.
  \label{eq:sum-ex2}
\end{align}
%----------------------------------------------------------------------
%----------------------------------------------------------------------
\paragraph{Rule 2: Positional disambiguation.}
When two indices range over the same set, as $i,j\in V$ do in
$NF_{ijmpr}$, suppressing one by name alone is ambiguous. A
dot~$(\,\cdot\,)$ marks the suppressed position:
$\overline{NF}_{\cdot jmpr}\coloneqq\sum_{i\in V}NF_{ijmpr}$
suppresses~$i$, while
$\overline{NF}_{i\cdot mpr}\coloneqq\sum_{j\in V}NF_{ijmpr}$
suppresses~$j$. This convention applies to all three rules below.

\medskip
\noindent\textbf{Rule~3
  \textnormal{(OR,~$\widecheck{\phantom{x}}$).}
  Binary root variables only.}
Suppressing one or more indices and writing a
$\widecheck{\phantom{x}}$ over the root variable symbol introduces a
new binary variable equal to~1 if and only if at least one root binary
sharing the same projected element~$\pi$ is active. For every
$\pi\in\Pi$ the new variable $\widecheck{\phi}_{\pi}$ is coupled to
the corresponding sum $\overline{\phi}_{\pi}$ (Rule~1) via:
\begin{align}
  \widecheck{\phi}_{\pi}
    &\leq \overline{\phi}_{\pi}
    && \forall\,\pi\in\Pi,
    \label{eq:or-lb} \\
  \overline{\phi}_{\pi}
    &\leq |\mathcal{S}|\;\widecheck{\phi}_{\pi}
    && \forall\,\pi\in\Pi,
    \label{eq:or-ub}
\end{align}
where $|\mathcal{S}|$ is the cardinality of the suppressed domain.
Together, \eqref{eq:or-lb}--\eqref{eq:or-ub} enforce
$\widecheck{\phi}_{\pi}=1 \iff \overline{\phi}_{\pi}\geq 1$ and,
equivalently, $\widecheck{\phi}_{\pi}=0 \iff
\overline{\phi}_{\pi}=0$.

\smallskip\noindent\textit{Example.}
From the binary $XF_{ijmpr}$, suppressing the indices $m, p, $ and $r$ we get 
($\Pi=E$, $\mathcal{S}=P\times\mathcal{M}\times R$,
$|\mathcal{S}|=|R||\mathcal{M}||P|$)
yields $\widecheck{XF}_{ij}$, which equals~1 whenever route
$(i,j)$ carries full RTIs:
\begin{align}
  \widecheck{XF}_{ij}
    &\leq \overline{XF}_{ij}
    && \forall\,(i,j)\in E,
    \label{eq:or-ex-lb} \\
  \overline{XF}_{ij}
    &\leq |R||\mathcal{M}||P|\;\widecheck{XF}_{ij}
    && \forall\,(i,j)\in E.
    \label{eq:or-ex-ub}
\end{align}

%----------------------------------------------------------------------
\medskip
\noindent\textbf{Rule~4
  \textnormal{(At-most-one,~$\widehat{\phantom{x}}$).}
  Binary root variables only.}
Suppressing one or more indices and writing a
$\widehat{\phantom{x}}$ over the root variable symbol introduces a
new binary variable that equals the sum and simultaneously caps it
at~1, asserting that at most one root binary sharing the same
projected element~$\pi$ may be active. For every $\pi\in\Pi$ the
derived variable is coupled to the corresponding sum via:
\begin{align}
  \widehat{\phi}_{\pi}
    &\leq \overline{\phi}_{\pi}
    && \forall\,\pi\in\Pi,
    \label{eq:amo-lb} \\
  \overline{\phi}_{\pi}
    &\leq \widehat{\phi}_{\pi}
    && \forall\,\pi\in\Pi.
    \label{eq:amo-ub}
\end{align}
Because $\widehat{\phi}_{\pi}\in\{0,1\}$, both inequalities together
force $\overline{\phi}_{\pi}\in\{0,1\}$: either no root binary is
active ($\widehat{\phi}_{\pi}=0$) or exactly one is
($\widehat{\phi}_{\pi}=1$). This is equivalent to an SOS1 constraint
on the group of root binaries indexed by~$\mathcal{S}$ that share the
same projected element~$\pi$.

\smallskip\noindent\textit{Example.}
From $XF_{ijmpr}$, suppressing~$m$ and~$r$
($\Pi=E\times P$, $\mathcal{S}=\mathcal{M}\times R$)
yields $\widehat{XF}_{ijp}$, which enforces that at most one
mode--RTI-type pair may carry product~$p$ on route $(i,j)$:
\begin{align}
  \widehat{XF}_{ijp}
    &\leq \overline{XF}_{ijp}
    && \forall\,(i,j)\in E,\;p\in P,
    \label{eq:amo-ex-lb} \\
  \overline{XF}_{ijp}
    &\leq \widehat{XF}_{ijp}
    && \forall\,(i,j)\in E,\;p\in P.
    \label{eq:amo-ex-ub}
\end{align}

\end{convention}
