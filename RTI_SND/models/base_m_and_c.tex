% ══════════════════════════════════════════════════════════════════
%  OBJECTIVE FUNCTION
% ══════════════════════════════════════════════════════════════════

\paragraph{Objective function.}
Minimise total daily transport cost across all lanes and modes,
plus the amortised purchase cost of any additional RTIs required:
%
\begin{align}
  \min \quad
  TC
  \;+\;
  \sum_{c\in C} p^{\text{RTI},c} \cdot P_{c}
  \label{eq:objective}
\end{align}

% ══════════════════════════════════════════════════════════════════
%  RTI TYPE SELECTION
% ══════════════════════════════════════════════════════════════════

\paragraph{RTI type selection.}
Each lane commits to exactly one RTI type for its full flows,
chosen from the admissible set $C_{ij}$:
%
\begin{align}
  \sum_{c \in C_{ij}} \widehat{XF}_{cij} &= 1
  && \forall\,(i,j) \in E
  \label{eq:type-selection}
\end{align}
\paragraph{RTI type selection.}
Each lane commits to exactly one RTI type for its full flows,
chosen from the admissible set $C_{ij}$:
%
\begin{align}
  \widehat{XF}_{ij} &= 1
  && \forall\,(i,j) \in E
  \label{eq:type-selection}
\end{align}


% ══════════════════════════════════════════════════════════════════
%  DEMAND SATISFACTION
% ══════════════════════════════════════════════════════════════════

\paragraph{Demand satisfaction.}
The total daily full-RTI flow of type $c$ on lane $(i,j)$,
summed over all modes, equals the type-specific demand if and
only if type $c$ is selected:
%
\begin{align}
  NF_{cij}
  &= \mu_{cij} \cdot \hat{X}_{cij}
  && \forall\,(i,j)\in E,\ c\in C_{ij}
  \label{eq:demand}
\end{align}

% ══════════════════════════════════════════════════════════════════
%  FLOW CONSERVATION
% ══════════════════════════════════════════════════════════════════

\paragraph{Flow conservation per RTI type.}
The closed-loop balance must hold separately for each type at every
node, regardless of whether the node is a hub or not:
%
\begin{align}
  \sum_{j\in V}\!\bigl(NF_{cij}+NI_{cij}+NE_{cij}\bigr)
  &=
  \sum_{j\in V}\!\bigl(NF^{c}_{ji}+NI^{c}_{ji}+NE^{c}_{ji}\bigr)
  && \forall\, i\in V,\ c\in C
  \label{eq:conservation}
\end{align}

% ══════════════════════════════════════════════════════════════════
%  INLAY RETURN FLOWS
% ══════════════════════════════════════════════════════════════════

\paragraph{Inlay return flows.}
Return inlays of type $c$ are sent back on the reverse lane in
proportion to the selected type-$c$ full shipments that required
them:
%
\begin{align}
  NI^{c}_{ji}
  &= ric_{cij} \cdot \mu_{cij} \cdot \hat{X}_{cij}
  && \forall\,(i,j)\in E,\ c\in C_{ij}
  \label{eq:inlay}
\end{align}

% ══════════════════════════════════════════════════════════════════
%  HUB STRUCTURE
% ══════════════════════════════════════════════════════════════════

\paragraph{Hub and non-hub node structure.}
Hub nodes $i \in H$ are unconstrained with respect to empty RTI
flows: they may simultaneously receive and dispatch empties of any
type, acting as consolidation and redistribution points.
Non-hub nodes $i \in V \setminus H$, by contrast, may not serve
as intermediate consolidation points for empty RTIs. For each
type $c$, a non-hub node is permitted either to \emph{receive}
empties (as a destination) or to \emph{dispatch} empties (as an
origin), but not both simultaneously.

The binaries $EI_{ci}$ and $EO_{ci}$ indicate whether
non-hub node $i$ has positive incoming or outgoing empty flows
of type $c$, using the dot notation of
Convention~\ref{conv:notation}:
%
\begin{align}
  NE^{c}_{\cdot i}
    &\leq M \cdot EI_{ci},
  \quad
  EI_{ci}
    \leq NE^{c}_{\cdot i}
  && \forall\, i \in V\setminus H,\ c \in C
  \label{eq:hub-EI-link}\\[4pt]
  NE^{c}_{i \cdot}
    &\leq M \cdot EO_{ci},
  \quad
  EO_{ci}
    \leq NE^{c}_{i \cdot}
  && \forall\, i \in V\setminus H,\ c \in C
  \label{eq:hub-EO-link}
\end{align}
%
where $NE^{c}_{\cdot i} = \sum_{j \in V}\sum_{m \in \mathcal{M}}
NE^{c}_{jim}$ and $NE^{c}_{i \cdot} = \sum_{j \in V}
\sum_{m \in \mathcal{M}} NE_{cijm}$ follow from
Convention~\ref{conv:notation}.
The mutual exclusion constraint then enforces the non-hub
restriction:
%
\begin{align}
  EI_{ci} + EO_{ci} &\leq 1
  && \forall\, i \in V\setminus H,\ c \in C
  \label{eq:non-hub}
\end{align}

% ══════════════════════════════════════════════════════════════════
%  FLOW--BINARY CONSISTENCY
% ══════════════════════════════════════════════════════════════════

\paragraph{Flow--binary consistency.}
Each type-mode binary is active if and only if the corresponding
daily flow is positive. The aggregated OR binaries follow from
Convention~\ref{conv:notation}:
%
\begin{align}
  NF_{cijm} &\leq M \cdot XF_{cijm},
  \quad XF_{cijm} \leq NF_{cijm}
  && \forall\,(i,j)\in E,\ c\in C,\ m\in\mathcal{M}
  \label{eq:binary-full}\\
  NI_{cijm} &\leq M \cdot XI_{cijm},
  \quad XI_{cijm} \leq NI_{cijm}
  && \forall\,(i,j)\in E,\ c\in C,\ m\in\mathcal{M}
  \label{eq:binary-inlay}\\
  NE_{cijm} &\leq M \cdot XE_{cijm},
  \quad XE_{cijm} \leq NE_{cijm}
  && \forall\,(i,j)\in E,\ c\in C,\ m\in\mathcal{M}
  \label{eq:binary-empty}
\end{align}

% ══════════════════════════════════════════════════════════════════
%  PER-SHIPMENT QUANTITIES
% ══════════════════════════════════════════════════════════════════

\paragraph{Per-shipment quantities.}
RTIs and total volume accumulated between dispatches are
proportional to the queue time. These are \emph{bilinear} and will
be linearised in Section~\ref{sec:linearisation}:
%
\begin{align}
  NS_{cijm}
  &= \bigl(NF_{cijm}+NI_{cijm}+NE_{cijm}\bigr)
     \cdot q_{ijm}
  && \forall\,(i,j)\in E,\ c\in C,\ m\in\mathcal{M}
  \label{eq:ship-N}\\
  VS_{ijm}
  &= \sum_{c\in C}\!\Bigl(
       NF_{cijm}\cdot v^{f,c}
      +NI_{cijm}\cdot v^{f,c}
      +NE_{cijm}\cdot v^{e,c}
     \Bigr)\cdot q_{ijm}
  && \forall\,(i,j)\in E,\ m\in\mathcal{M}
  \label{eq:ship-vol}
\end{align}
% Add after eq:vol-bounds, as a new paragraph:
\paragraph{Queue time bounds.}
The queue time on each lane is bounded below and above by
mode-specific limits, and is only meaningful when the mode is
active:
%
\begin{align}
  q^{\min}_{m}\cdot\overline{X}_{ijm}
  \;\leq\; q_{ijm}
  \;\leq\; q^{\max}_{m}\cdot\overline{X}_{ijm}
  && \forall\,(i,j)\in E,\ m\in\mathcal{M}
  \label{eq:q-bounds}
\end{align}
% ══════════════════════════════════════════════════════════════════
%  SHIPMENT VOLUME BOUNDS
% ══════════════════════════════════════════════════════════════════

\paragraph{Shipment volume bounds.}
Each active mode-lane combination must respect a minimum load and
a maximum vehicle capacity:
%
\begin{align}
  \omega_{ijm}\cdot\overline{X}_{ijm}
  \;\leq\; VS_{ijm}
  \;\leq\; \delta_{ijm}\cdot\overline{X}_{ijm}
  && \forall\,(i,j)\in E,\ m\in\mathcal{M}
  \label{eq:vol-bounds}
\end{align}

% ══════════════════════════════════════════════════════════════════
%  RTI POOL REQUIREMENTS
% ══════════════════════════════════════════════════════════════════

\paragraph{Total RTI pool requirements.}
The RTIs allocated to each lane must cover queue stock,
in-transit stock, safety stock at both endpoints, and a
maintenance loss margin:
%
\begin{align}
  TNCF_{cij}
  &= (1+\alpha^{f})\!\left(
       NS_{cij}
       + l_{ij}\cdot NF_{cij}
       + \overline{XF}_{cij}
         \bigl(ssf^{i,c}_{ij}+ssf^{j,c}_{ij}\bigr)
     \right)
  && \forall\,(i,j)\in E,\ c\in C
  \label{eq:pool-full}\\[4pt]
  TNCE_{cij}
  &= (1+\alpha^{e})\!\left(
       NS_{cij}
       + l_{ij}\cdot NE_{cij}
       + \overline{XE}_{cij}
         \bigl(sse^{i,c}_{ij}+sse^{j,c}_{ij}\bigr)
     \right)
  && \forall\,(i,j)\in E,\ c\in C
  \label{eq:pool-empty}\\[4pt]
  TNCI_{cij}
  &= (1+\alpha^{f})\!\left(
       NS_{cij}
       + l_{ij}\cdot NI_{cij}
     \right)
  && \forall\,(i,j)\in E,\ c\in C
  \label{eq:pool-inlay}
\end{align}

% ══════════════════════════════════════════════════════════════════
%  RTI PURCHASE
% ══════════════════════════════════════════════════════════════════

\paragraph{RTI purchase.}
For each type the network purchases any shortfall between the total
pool requirement and the existing stock; a surplus remains idle:
%
\begin{align}
  \sum_{(i,j)\in E}\!\bigl(
    TNCF_{cij}+TNCE_{cij}+TNCI_{cij}
  \bigr)
  &\leq s^{\text{RTI},c} + P_{c}
  && \forall\, c\in C,\quad P_{c}\geq 0
  \label{eq:purchase}
\end{align}

% ══════════════════════════════════════════════════════════════════
%  DAILY TRANSPORT COSTS
% ══════════════════════════════════════════════════════════════════

\paragraph{Daily transport costs.}
The per-shipment cost combines a volumetric charge and a fixed
dispatch cost. Dividing by the queue time converts it to a daily
equivalent; this division is \emph{bilinear} and will be
linearised in Section~\ref{sec:linearisation}:
%
\begin{align}
  TCS_{ijm}
  &= VS_{ijm}\cdot c_{ijm}
     + \overline{X}_{ijm}\cdot f_{ijm}
  && \forall\,(i,j)\in E,\ m\in\mathcal{M}
  \label{eq:cost-ship}\\[4pt]
  TC_{ijm}\cdot q_{ijm}
  &= TCS_{ijm}
  && \forall\,(i,j)\in E,\ m\in\mathcal{M}
  \label{eq:cost-daily}
\end{align}
