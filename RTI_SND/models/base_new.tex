% ══════════════════════════════════════════════════════════════════
%  OBJECTIVE FUNCTION
% ══════════════════════════════════════════════════════════════════

\paragraph{Objective function.}
Minimise total daily transport cost across all routes,
plus the amortised purchase cost of any additional RTIs required:
%
\begin{align}
  \min \quad
  TC
  \;+\;
  \sum_{r\in R} p^{\text{RTI}}_r \cdot P_r + CH
  \label{eq:objective}
\end{align}

% ══════════════════════════════════════════════════════════════════
%  FLOW CONSERVATION
% ══════════════════════════════════════════════════════════════════

\paragraph{Flow conservation.}
The closed-loop balance must hold at every node:
%
\begin{align}
  N_{i\cdot r} &=
N_{\cdot i r}
  && \forall\, i\in V
  \label{eq:conservation}
\end{align}

\paragraph{Flow--binary consistency.}
Each binary is active if and only if the corresponding daily flow
is positive:
%
\begin{align}
  NF_{ijmpr} &\leq M \cdot XF_{ijmpr} \quad XF_{ijmpr} \leq NF_{ijmpr} \cdot bf_{ijmpr}
  && \forall\,(i,j)\in E \label{eq:binary-full}\\
  NI_{ijmpr} &\leq M \cdot XI_{ijmpr} \quad XI_{ijmpr} \leq NI_{ijmpr} \cdot bi_{ijmpr}
  && \forall\,(i,j)\in E \label{eq:binary-inlay}\\
  NE_{ijmpr} &\leq M \cdot XE_{ijmpr} \quad XE_{ijmpr} \leq NE_{ijmpr} \cdot be_{ijmpr}
  && \forall\,(i,j)\in E \label{eq:binary-empty}
\end{align}

% ══════════════════════════════════════════════════════════════════
%  ROUTE ACTIVATION
% ══════════════════════════════════════════════════════════════════

\paragraph{Route activation.}
A route is active if and only if it carries full, empty, or inlay RTIs:
%
\begin{align}
  3\cdot X_{ijmpr} &\geq XF_{ijmpr}+XE_{ijmpr}+XI_{ijmpr}
  && \forall\,(i,j)\in E \\
  X_{ijmpr} &\leq XF_{ijmpr}+XE_{ijmpr}+XI_{ijmpr}
  && \forall\,(i,j)\in E
\end{align}

% ══════════════════════════════════════════════════════════════════
%  RTI TYPE SELECTION
% ══════════════════════════════════════════════════════════════════

\paragraph{Full Routes activation}
A route carries full RTIs only if average daily demand is positive:
%
\begin{align}
  \widehat{XF}_{ijp} & =  bp_{ijp}
  && \forall\,(i,j)\in E 
  \label{eq:full-activation}
\end{align}

\paragraph{Extension M --- Transport Modes.}

\begin{align}
  \overline{X}_{ijm} \leq bm_{ijm}
  && \forall\,(i,j)\in E,\;{m\in\mathcal{M}}
  \label{eq:ext-m-or}
\end{align}
\begin{align}
  \sum_{m\in \mathcal{M}}\overline{X}_{ijm} \leq 1
  && {\forall\,(i,j)\in E}
  \label{eq:at-most-one-m}
\end{align}
{where $\widehat{X}_{ij}$ is the at-most-one aggregate binary
(Rule~4) equal to~1 if exactly one mode is active on route $(i,j)$.}

\paragraph{Hub Nodes.}
{Hub nodes $i \in H \subseteq V$ are intermediate
consolidation points that aggregate outgoing flows and incur a handling cost.
The parameter $fhc_i \geq 0$ is the fixed cost of activating hub $i$, and
$vhc_i \geq 0$ is the variable cost per unit volume handled at hub $i$.}

Hub nodes $i\in H$ are unconstrained. At every non-hub node
$i\in V\setminus H$ the incoming and outgoing empty-flow binaries are linked
to aggregated flows via Rule~3 and forced mutually exclusive. %
\begin{align}
  \overline{XE}_{\cdot ir} + \overline{XE}_{i \cdot r} \leq 1
  && \forall\,i\in V\setminus H
  \label{eq:ext-h-excl}
\end{align}
%
{For every hub node $i \in H$, the following equations define
the hub activation binary $XH_i$, the total volume $VH_i$ flowing through
hub $i$, and the resulting hub cost $CH_i$:}
%
\begin{align}
  XH_i &= \overline{X}_{i\cdot}
  && {\forall\,i\in H}
  \label{eq:ext-h-XH} \\[4pt]
  VH_i &= V_{i\cdot}
  && {\forall\,i\in H}
  \label{eq:ext-h-VH} \\[4pt]
  CH_{i} &= fhc_{i}\cdot XH_i + vhc_i \cdot VH_i
  && {\forall\,i\in H}
  \label{eq:ext-h-CH}
\end{align}
%
{where $\overline{X}_{i\cdot} \coloneqq \sum_{j\in V}
\overline{X}_{ij}$ and $V_{i\cdot} \coloneqq \sum_{j\in V} V_{ij}$ are the
aggregated outgoing activation binary and total volume at node $i$.


% ── Extension P ──────────────────────────────────────────────────────────────
% ── Extension R ──────────────────────────────────────────────────────────────
% ══════════════════════════════════════════════════════════════════
%  DEMAND SATISFACTION
% ══════════════════════════════════════════════════════════════════

\paragraph{Full RTIs route flow}
The daily full-RTI flow on route $(i,j)$ equals demand if and only
if the route is active:
%
\begin{align}
  NF_{ijmpr} &= \mu_{ijmpr} \cdot X_{ijmpr}
  && \forall\,(i,j)\in E
  \label{eq:demand}
\end{align}

% ══════════════════════════════════════════════════════════════════
%  INLAY RETURN FLOWS
% ══════════════════════════════════════════════════════════════════

\paragraph{Inlay return flows.}
Return inlays are sent back on the reverse route in proportion to
the full shipments that required them:
%
\begin{align}
  NI_{ji} &= ric_{ijmpr} \cdot \mu_{ijmpr} \cdot X_{ijmpr}
  && \forall\,(i,j)\in E
  \label{eq:inlay}
\end{align}

% ══════════════════════════════════════════════════════════════════
%  FLOW–BINARY CONSISTENCY
% ══════════════════════════════════════════════════════════════════

\paragraph{Daily average volumes.}
Full and inlay RTIs occupy volume $v^{f}$; empty RTIs occupy $v^{e}$:
%
\begin{align}
  VF_{ijmpr} &= NF_{ijmpr} \cdot v^f
  && \forall\,(i,j)\in E \label{eq:volume-F}\\
  VI_{ijmpr} &= NI_{ijmpr} \cdot v^f
  && \forall\,(i,j)\in E \label{eq:volume-I}\\
  VE_{ijmpr} &= NE_{ijmpr} \cdot v^e
  && \forall\,(i,j)\in E \label{eq:volume-E}\\
  V_{ijmpr}  &= VF_{ijmpr} + VI_{ijmpr} + VE_{ijmpr}
  && \forall\,(i,j)\in E \label{eq:volume-daily}
\end{align}

\paragraph{Per-shipment RTI counts.}
RTIs accumulated between dispatches are proportional to the queue
time. These are \emph{nonlinear} and will be linearised in
Section~\ref{sec:linearisation}:
%
\begin{align}
  NSF_{ijmpr} &= NF_{ijmpr} \cdot q_{ijmpr}
  && \forall\,(i,j)\in E \label{eq:ship-NF}\\
  NSI_{ijmpr} &= NI_{ijmpr} \cdot q_{ijmpr}
  && \forall\,(i,j)\in E \label{eq:ship-NI}\\
  NSE_{ijmpr} &= NE_{ijmpr} \cdot q_{ijmpr}
  && \forall\,(i,j)\in E \label{eq:ship-NE}\\
  NS_{ijmpr}  &= NSF_{ijmpr} + NSI_{ijmpr} + NSE_{ijmpr}
  && \forall\,(i,j)\in E \label{eq:ship-N}
\end{align}

\paragraph{Per-shipment volumes.}
Full and inlay RTIs occupy volume $v^{f}$; empty RTIs occupy $v^{e}$:
%
\begin{align}
  VSF_{ijmpr} &= VF_{ijmpr} \cdot q_{ijmpr}
  && \forall\,(i,j)\in E \label{eq:ship-VF}\\
  VSI_{ijmpr} &= VI_{ijmpr} \cdot q_{ijmpr}
  && \forall\,(i,j)\in E \label{eq:ship-VI}\\
  VSE_{ijmpr} &= VE_{ijmpr} \cdot q_{ijmpr}
  && \forall\,(i,j)\in E \label{eq:ship-VE}\\
  VS_{ijmpr}  &= VSF_{ijmpr} + VSI_{ijmpr} + VSE_{ijmpr}
  && \forall\,(i,j)\in E \label{eq:ship-vol}
\end{align}

% ══════════════════════════════════════════════════════════════════
%  QUEUE TIME BOUNDS
% ══════════════════════════════════════════════════════════════════

\paragraph{Queue time bounds.}
The queue time is bounded and only meaningful when the route is active:
%
\begin{align}
q^{\min}_m \cdot \overline{X}_{ijm}
  \;\leq\; q_{ijm}
\;\leq\; q^{\max}_m \cdot \overline{X}_{ijm}
  && \forall\,(i,j)\in E
  \label{eq:q-bounds}
\end{align}

% ══════════════════════════════════════════════════════════════════
%  SHIPMENT VOLUME BOUNDS
% ══════════════════════════════════════════════════════════════════

\paragraph{Shipment volume bounds.}
Each active route must respect a minimum load and a maximum
vehicle capacity:
%
\begin{align}
  \omega_{ijm} \cdot \overline{X}_{ijm}
  \;\leq\; VS_{ijm}
  \;\leq\; \delta_{ijm} \cdot \overline{X}_{ijm}
  && \forall\,(i,j)\in E
  \label{eq:vol-bounds}
\end{align}

% ══════════════════════════════════════════════════════════════════
%  RTI POOL REQUIREMENTS
% ══════════════════════════════════════════════════════════════════

\paragraph{Total RTI pool requirements.}
The RTIs allocated to each route cover queue stock, in-transit
stock, safety stock at both endpoints, and a maintenance loss margin:
%
\begin{align}
  TNCF_{ijmpr}
  &= (1+\alpha^{f})\!\left(
       NSF_{ijmpr}
       + l_{ijmpr}\cdot NF_{ijmpr}
       + XF_{ijmpr}\bigl(ssf^{i}_{ijpr}+ssf^{j}_{ijpr}\bigr)
     \right)
  && \forall\,(i,j)\in E
  \label{eq:pool-full}\\[4pt]
  TNCE_{ijmpr}
  &= (1+\alpha^{e})\!\left(
       NSE_{ijmpr}
       + l_{ijmpr}\cdot NE_{ijmpr}
       + XE_{ijmpr}\bigl(sse^{i}_{ijpr}+sse^{j}_{ijpr}\bigr)
     \right)
  && \forall\,(i,j)\in E
  \label{eq:pool-empty}\\[4pt]
  TNCI_{ijmpr}
  &= (1+\alpha^{f})\!\left(
       NSI_{ijmpr}
       + l_{ijmpr}\cdot NI_{ijmpr}
     \right)
  && \forall\,(i,j)\in E
  \label{eq:pool-inlay}\\[4pt]
  TNC_{ijmpr}
  &= TNCF_{ijmpr} + TNCE_{ijmpr} + TNCI_{ijmpr}
  && \forall\,(i,j)\in E
  \label{eq:pool-total}
\end{align}

% ══════════════════════════════════════════════════════════════════
%  RTI PURCHASE
% ══════════════════════════════════════════════════════════════════

\paragraph{RTI purchase.}
The network purchases any shortfall between the total pool
requirement and the existing stock:
%
\begin{align}
  TNC_{p}
  &\leq s^{\text{RTI}}_p + P_p,
  \qquad P_p \geq 0
  \label{eq:purchase}
\end{align}

% ══════════════════════════════════════════════════════════════════
%  DAILY TRANSPORT COSTS
% ══════════════════════════════════════════════════════════════════

\paragraph{Daily transport costs.}
The per-shipment cost combines a volumetric charge and a fixed
dispatch cost. Dividing by the queue time converts it to a daily
equivalent; this division is \emph{nonlinear} and will be
linearised in Section~\ref{sec:linearisation}:
%
\begin{align}
  TCS_{ijm}
  &= VS_{ijm} \cdot c_{ijm}
     + X_{ijm} \cdot f_{ijm}
  && \forall\,(i,j)\in E
  \label{eq:cost-ship}\\[4pt]
  TC_{ijm} \cdot q_{ijm}
  &= TCS_{ijm}
  && \forall\,(i,j)\in E
  \label{eq:cost-daily}
\end{align}

\subsection{Model variations}
