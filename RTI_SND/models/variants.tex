\section{Model Extensions}
\label{sec:extensions}

The base model can be extended by activating any subset of five independent
additions: \textbf{M} (transport modes), \textbf{R} (RTI types),
\textbf{H} (hub nodes), \textbf{P} (explicit product flows), and \textbf{L}
(linearisation of the queue-time products).
Each extension is self-contained and any combination
is valid.

Extensions M, R, H, and P each introduce a new set with its index, replacing the root variables of the model by adding and additional index. Extension~L is qualitatively different: it does
not add an index to existing variables but instead discretises the
continuous queue time $q_{ij}$ over a finite set $\mathcal{Q}_{ij}$ of
admissible integer values in order to discritze teh model. The result
is a pure MILP formulation with no nonlinear terms.

Each extension operates through two mechanisms.

\begin{enumerate}
  \item \textbf{Index addition or variable introduction.}
        Table~\ref{tab:index-map} lists, for every symbol, which extensions
        add an index to the variable or parameter or introduce a new symbol (\textbf{New}), either a variable or parameter.
        Symbols marked $\circ$ in a column are unaffected; symbols marked
        $\times$ are replaced by derived expressions
        (Section~\ref{sec:ext-new-constraints}).

  \item \textbf{Constraint transformation.}
        Table~\ref{tab:constraint-transform} records, for every base
        constraint, what each active extension does to it. Extensions not
        listed for a constraint leave it unchanged ($\circ$). Unless specified, a constraint is transformed automatically if a variable is tranformed that appears in the constraint by iterating over the new set, that is if an indexed is added to a variable or parameter.
\end{enumerate}

\subsection{New Constraints Introduced by Each Extension}
\label{sec:ext-new-constraints}

\paragraph{Extension M.}
Only one type of transport mode can for the same route.
%
\begin{align}
  \overline{X}_{ijm} \leq bm_{ijm} && \forall\,(i,j)\in E  \label{eq:ext-m-or}
\end{align}
Equation \ref{eq:route-enforced-by-parameters} is replaced by \ref{eq:at-most-one-m} to enforce only one type of transport mode per route.
\begin{align}
  \widehat{X}_{ij} \leq bf_{ij}+be_{ij}+bi_{ij} && \forall\,(i,j)\in E  \label{eq:at-most-one-m}
\end{align}


\paragraph{Extension R.}
Each route commits to exactly one admissible RTI type (Rule~4); demand and
inlay flows are conditioned on the selected type:
Equation \ref{eq:full-activation} is replaced by \ref{eq:or-for-r} to enforce only different full RTIs types are sent.
%
\begin{align}
  \overline{XF}_{ijr} &= bf_{ijr}
  && \forall\,(i,j)\in E , r \in R
  \label{eq:or-for-r} 
\end{align}
%
 
Equation \ref{eq:route-enforced-by-parameters} is replaced by \ref{eq:at-most-one-m} to enforce only one type of transport mode per route.
\begin{align}
  \overline{X}_{ij} \leq \sum_{r\in R} bf_{ijr}+be_{ijr}+bi_{ijr} && \forall\,(i,j)\in E  \label{eq:link-r-and-route}
\end{align}
\paragraph{Extension H.}
Hub nodes $i\in H$ are unconstrained. At every non-hub node
$i\in V\setminus H$ the incoming and outgoing empty-flow binaries are linked
to aggregated flows via Rule~3 and forced mutually exclusive. The index $r$
appears only when Ext.~R is also active:
%
\begin{align}
  \overline{XE}_{\cdot i} + \overline{XE}_{i \cdot} &\leq 1
  && \forall\,i\in V\setminus H
  \label{eq:ext-h-excl}
\end{align}


\begin{align}
  XH_i = \overline{X}_{i\cdot}\\
  VH_i = V_{i \cdot} \\
  CH_{i} = fhc_{i}\cdot  XH_i + VH_i \cdot vhc_i 
\end{align}
 
CH is to be added to the monetary costs. 

\paragraph{Extension P.}
%
%
\begin{align}
  NF_{ijpr} =  \mu_{ijpr}*XF_{ijpr} \\ 
  \mu_{ijpr} = \frac{d_{ijp}}{\eta_{pr}}
\end{align}
%
\begin{align}
  \widehat{XF}_{ijp} &= bf_{ijp}
  && \forall\,(i,j)\in E , r \in R
  \label{eq:ext-r-sel} 
\end{align}
\paragraph{Extension L — Linearisation of products of two decision variables.}

The base model contains three families of terms that are products of two
decision variables, all of the form $(\text{flow})\times q_{ij}$:
\begin{enumerate}
  \item $NF_{ij}\cdot q_{ij}$ and $NI_{ij}\cdot q_{ij}$, which involve
        flows fixed by demand once the route is active.
  \item $NE_{ij}\cdot q_{ij}$, which involves a free flow variable.
  \item $TC_{ij}\cdot q_{ij}$, which equates per-shipment to daily cost.
\end{enumerate}
%
Extension~L eliminates all three by discretising $q_{ij}$ over the finite set
$\mathcal{Q}_{ij}\subset\mathbb{Z}_{>0}$ of admissible queue times.
A binary variable $z_{ij,k}$ selects the active queue time value. If
Ext.~M is also active, the index $m$ is appended everywhere
($z_{ijm,k}$, $\mathcal{Q}_{ijm}$, etc.).

\medskip\noindent\textbf{Queue-time discretisation.}
The queue time is expressed as a weighted sum over admissible integer values
and deactivated when the route is idle:
%
\begin{align}
  q_{ij} &= \sum_{k\in\mathcal{Q}_{ij}} k\cdot z_{ij,k}
  && \forall\,(i,j)\in E
  \label{eq:ext-l-q} \\[4pt]
  \sum_{k\in\mathcal{Q}_{ij}} z_{ij,k} &= X_{ij}
  && \forall\,(i,j)\in E
  \label{eq:ext-l-z-sum}
\end{align}
%
Constraint \eqref{eq:ext-l-z-sum} ensures exactly one queue-time value is
selected when the route is active and none when it is not.
Together \eqref{eq:ext-l-q}--\eqref{eq:ext-l-z-sum} replace the bounds
\eqref{eq:q-bounds}.

\medskip\noindent\textbf{Linearisation of family 1 (fixed-demand flows).}
Because $NF_{ij}=\mu_{ij}\cdot X_{ij}$ and \eqref{eq:ext-l-z-sum} implies
$X_{ij}\cdot z_{ij,k}=z_{ij,k}$, the products reduce directly to linear
expressions without auxiliary variables:
%
\begin{align}
  NSF_{ij} &= \mu_{ij}\!\sum_{k\in\mathcal{Q}_{ij}} k\cdot z_{ij,k}
  && \forall\,(i,j)\in E
  \label{eq:ext-l-NSF} \\[4pt]
  NSI_{ij} &= ric_{ij}\cdot\mu_{ij}\!\sum_{k\in\mathcal{Q}_{ij}} k\cdot z_{ij,k}
  && \forall\,(i,j)\in E
  \label{eq:ext-l-NSI}
\end{align}

\medskip\noindent\textbf{Linearisation of family 2 (free empty-RTI flow).}
Introduce the auxiliary continuous variable
$\widetilde{NE}_{ij,k}\coloneqq NE_{ij}\cdot z_{ij,k}$, linearised by the
standard McCormick reformulation for a product of a continuous and a binary
variable:
%
\begin{align}
  \widetilde{NE}_{ij,k} &\leq NE^{\max}_{ij}\cdot z_{ij,k}
  && \forall\,(i,j)\in E,\; k\in\mathcal{Q}_{ij}
  \label{eq:ext-l-NEt-ub1} \\[2pt]
  \widetilde{NE}_{ij,k} &\leq NE_{ij}
  && \forall\,(i,j)\in E,\; k\in\mathcal{Q}_{ij}
  \label{eq:ext-l-NEt-ub2} \\[2pt]
  \widetilde{NE}_{ij,k} &\geq NE_{ij} - NE^{\max}_{ij}(1-z_{ij,k})
  && \forall\,(i,j)\in E,\; k\in\mathcal{Q}_{ij}
  \label{eq:ext-l-NEt-lb} \\[2pt]
  \widetilde{NE}_{ij,k} &\geq 0
  && \forall\,(i,j)\in E,\; k\in\mathcal{Q}_{ij}
  \label{eq:ext-l-NEt-nn}
\end{align}
%
where $NE^{\max}_{ij}$ is a valid upper bound on $NE_{ij}$.
The per-shipment count and volume then become:
%
\begin{align}
  NSE_{ij} &= \sum_{k\in\mathcal{Q}_{ij}} k\cdot\widetilde{NE}_{ij,k}
  && \forall\,(i,j)\in E
  \label{eq:ext-l-NSE} \\[4pt]
  VSE_{ij} &= v^{e}\!\sum_{k\in\mathcal{Q}_{ij}} k\cdot\widetilde{NE}_{ij,k}
  && \forall\,(i,j)\in E
  \label{eq:ext-l-VSE}
\end{align}

\medskip\noindent\textbf{Linearisation of family 3 (daily cost ratio).}
Introduce the auxiliary continuous variable
$\widetilde{TC}_{ij,k}\coloneqq TC_{ij}\cdot z_{ij,k}$, linearised
analogously with an upper bound $TC^{\max}_{ij}$:
%
\begin{align}
  \widetilde{TC}_{ij,k} &\leq TC^{\max}_{ij}\cdot z_{ij,k}
  && \forall\,(i,j)\in E,\; k\in\mathcal{Q}_{ij}
  \label{eq:ext-l-TCt-ub1} \\[2pt]
  \widetilde{TC}_{ij,k} &\leq TC_{ij}
  && \forall\,(i,j)\in E,\; k\in\mathcal{Q}_{ij}
  \label{eq:ext-l-TCt-ub2} \\[2pt]
  \widetilde{TC}_{ij,k} &\geq TC_{ij} - TC^{\max}_{ij}(1-z_{ij,k})
  && \forall\,(i,j)\in E,\; k\in\mathcal{Q}_{ij}
  \label{eq:ext-l-TCt-lb} \\[2pt]
  \widetilde{TC}_{ij,k} &\geq 0
  && \forall\,(i,j)\in E,\; k\in\mathcal{Q}_{ij}
  \label{eq:ext-l-TCt-nn}
\end{align}
%
The daily-cost constraint \eqref{eq:cost-daily} is then replaced by:
%
\begin{align}
  \sum_{k\in\mathcal{Q}_{ij}} k\cdot\widetilde{TC}_{ij,k} &= TCS_{ij}
  && \forall\,(i,j)\in E
  \label{eq:ext-l-cost-daily}
\end{align}
%
After applying Extension~L the model contains no products of two decision
variables and is a pure mixed-integer linear programme. If Ext.~M is also
active, all of \eqref{eq:ext-l-q}--\eqref{eq:ext-l-cost-daily} are
replicated $\forall m\in\mathcal{M}$ with mode-indexed counterparts.
