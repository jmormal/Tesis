%  CONVENTIONS

\begin{convention}[Index suppression and binary aggregation]
\label{conv:notation}
All variables are defined at the finest level $(i,j,c,m)$.

\medskip
\noindent\textbf{1. Continuous and integer variables} ---
suppression denotes summation:
\begin{align*}
  \phi_{cij} &\;\coloneqq\; \sum_{m\in\mathcal{M}} \phi_{cijm},
  &
  \phi_{ij} &\;\coloneqq\;
    \sum_{c\in C}\sum_{m\in\mathcal{M}} \phi_{cijm}
\end{align*}

\medskip
\noindent\textbf{2. Positional dot for same-set indices} ---
when both $i$ and $j$ belong to $V$, a dot marks the suppressed
position:
\begin{align*}
  \phi_{c}_{\cdot jm} &\;\coloneqq\; \sum_{i\in V}\phi_{cijm}
    &&\text{(sum over origin)},\\
  \phi_{c}_{i\cdot m} &\;\coloneqq\; \sum_{j\in V}\phi_{cijm}
    &&\text{(sum over destination)}
\end{align*}

\medskip
\noindent\textbf{3. Binary aggregation via CP-SAT primitives}:
\begin{align*}
  \overline{XF}_{cij}=1
    &\;\Leftrightarrow\;
    \texttt{bool\_or}(XF_{cijm},\;m\in\mathcal{M}) \\
  \overline{XE}_{cij}=1
    &\;\Leftrightarrow\;
    \texttt{bool\_or}(XE_{cijm},\;m\in\mathcal{M}) \\
  \overline{X}_{ijm}=1
    &\;\Leftrightarrow\;
    \texttt{bool\_or}(XF_{cijm},XI_{cijm},XE_{cijm},\;c\in C)
\end{align*}
\end{convention}

\begin{convention}[Generalised disjunctive programming notation]
\label{conv:gdp}
The model is expressed using Generalised Disjunctive Programming
(GDP) notation \citep{Raman1994,Grossmann2012}. Each disjunction
identifies a \emph{leader binary} that activates an entire block
of dependent variables and constraints simultaneously.

\medskip
\noindent\textbf{Exclusive disjunction} ($\veebar$, exactly one
term active):
\begin{equation*}
  \veebar_{k\in K}
  \begin{bmatrix}
    Y_k \\ g_k(x) \leq 0
  \end{bmatrix}
  \quad\equiv\quad
  \texttt{exactly\_one}(Y_k,\;k\in K)
  \;\;\text{and}\;\;
  Y_k=1\Rightarrow g_k(x)\leq 0
\end{equation*}

\noindent\textbf{Independent disjunction} ($\vee$, any subset
active):
\begin{equation*}
  \vee_{k\in K}
  \begin{bmatrix}
    Y_k \\ g_k(x) \leq 0
  \end{bmatrix}
  \quad\equiv\quad
  Y_k=1\Rightarrow g_k(x)\leq 0
  \;\;\forall\,k\in K
\end{equation*}

\noindent In both cases the leader binary appears on the first row
of the block; all remaining rows are constraints enforced only
when that leader is active. Inactive leaders force all dependent
variables in the block to zero. Since all flow variables are
non-negative integers, strict positivity $\phi>0$ is written
$\phi\geq 1$ throughout.
\end{convention}

% ══════════════════════════════════════════════════════════════════
%  OBJECTIVE FUNCTION
% ══════════════════════════════════════════════════════════════════

\paragraph{Objective function.}
Minimise total daily transport cost plus the purchase cost of any
additional RTIs required:
%
\begin{align}
  \min \quad
  \sum_{(i,j)\in E}\sum_{m\in\mathcal{M}} TC_{ijm}
  \;+\;
  \sum_{c\in C} p^{\text{RTI},c} \cdot P_{c}
  \label{eq:objective}
\end{align}

% ══════════════════════════════════════════════════════════════════
%  DISJUNCTION 1 — TYPE SELECTION
% ══════════════════════════════════════════════════════════════════

\paragraph{Disjunction 1 — RTI type selection.}
Each lane activates exactly one RTI type. The selected type fixes
the full and inlay flows; the rejected types are forced to zero.
This is an exclusive disjunction over $C_{ij}$:
%
\begin{equation}
  \veebar_{c\,\in\,C_{ij}}
  \begin{bmatrix}
    \hat{X}_{cij} \\[4pt]
    NF_{cij} = \mu_{cij} \\[2pt]
    NI_{c}_{ji} = ric_{cij}\cdot\mu_{cij}
  \end{bmatrix}
  \qquad \forall\,(i,j)\in E
  \label{eq:disj-type}
\end{equation}

% ══════════════════════════════════════════════════════════════════
%  FLOW CONSERVATION
% ══════════════════════════════════════════════════════════════════

\paragraph{Flow conservation per RTI type.}
These constraints are unconditional and hold at every node for
every type regardless of which disjunctive branch is active:
%
\begin{align}
  \sum_{j\in V}\!\bigl(NF_{cij}+NI_{cij}+NE_{cij}\bigr)
  &=
  \sum_{j\in V}\!\bigl(NF_{c}_{ji}+NI_{c}_{ji}+NE_{c}_{ji}\bigr)
  && \forall\, i\in V,\ c\in C
  \label{eq:conservation}
\end{align}

% ══════════════════════════════════════════════════════════════════
%  DISJUNCTION 2 — EMPTY FLOW ACTIVATION
% ══════════════════════════════════════════════════════════════════

\paragraph{Disjunction 2 — Empty RTI flow activation.}
For each lane-type-mode triple, the empty-flow leader $XE^c_{ijm}$
activates the flow variable and its contribution to the pool.
This is an independent disjunction (any subset of
$(c,m)$ combinations may be active simultaneously):
%
\begin{equation}
  \vee_{(c,m)}
  \begin{bmatrix}
    XE_{cijm} \\[4pt]
    XE_{cijm} \;\doteq\; (NE_{cijm} \geq 1) \\[2pt]
    TNCE_{cij} = (1+\alpha^{e})\!\left(
      NS_{cij}
      + l_{ij}\cdot NE_{cij}
      + \overline{XE}_{cij}
        \bigl(sse^{i,c}_{ij}+sse^{j,c}_{ij}\bigr)
    \right)
  \end{bmatrix}
  \;\forall\,(i,j)\in E
  \label{eq:disj-empty}
\end{equation}

% ══════════════════════════════════════════════════════════════════
%  HUB STRUCTURE
% ══════════════════════════════════════════════════════════════════

\paragraph{Hub and non-hub node structure.}
At non-hub nodes $i\in V\setminus H$ the incoming and outgoing
empty-flow leaders are mutually exclusive. Hub nodes $i\in H$
are unconstrained:
%
\begin{align}
  EI_{ci}
    &\;\doteq\; (NE_{c}_{\cdot i} \geq 1)
  && \forall\, i\in V\setminus H,\ c\in C
  \label{eq:hub-EI}\\
  EO_{ci}
    &\;\doteq\; (NE_{c}_{i\cdot} \geq 1)
  && \forall\, i\in V\setminus H,\ c\in C
  \label{eq:hub-EO}\\
  \texttt{at\_most\_one}&\bigl(EI_{ci},\,EO_{ci}\bigr)
  && \forall\, i\in V\setminus H,\ c\in C
  \label{eq:non-hub}
\end{align}

% ══════════════════════════════════════════════════════════════════
%  DISJUNCTION 3 — SHIPMENT ACTIVATION
% ══════════════════════════════════════════════════════════════════

\paragraph{Disjunction 3 — Shipment activation.}
The mode-level leader $\overline{X}_{ijm}$ activates the entire
shipment block: queue time, volume, per-shipment RTI counts, and
transport costs. When inactive, all these quantities are zero.
This is an independent disjunction over $(i,j,m)$:
%
\begin{equation}
  \vee_{(i,j,m)}
  \begin{bmatrix}
    \overline{X}_{ijm} \\[4pt]
    q_{ijm} \in [q^{\min}_{m},\; q^{\max}_{m}] \\[4pt]
    NS_{cijm} =
      \bigl(NF_{cijm}+NI_{cijm}+NE_{cijm}\bigr)
      \cdot q_{ijm}
      \hspace{8pt} \forall\,c\in C \\[4pt]
    VS_{ijm} =
      \displaystyle\sum_{c\in C}\!\Bigl(
        NF_{cijm}\cdot v^{f,c}
       +NI_{cijm}\cdot v^{f,c}
       +NE_{cijm}\cdot v^{e,c}
      \Bigr)\cdot q_{ijm} \\[6pt]
    VS_{ijm} \in [\omega_{ijm},\;\delta_{ijm}] \\[4pt]
    TCS_{ijm} = VS_{ijm}\cdot c_{ijm}
      + \overline{X}_{ijm}\cdot f_{ijm} \\[4pt]
    TC_{ijm}\cdot q_{ijm} = TCS_{ijm}
  \end{bmatrix}
  \label{eq:disj-ship}
\end{equation}

Note: the per-shipment products $NS_{cijm}$, $VS_{ijm}$, and
$TC_{ijm}\cdot q_{ijm}$ are \emph{bilinear} in the flow variables
and $q_{ijm}$; they will be linearised in
Section~\ref{sec:linearisation}.

% ══════════════════════════════════════════════════════════════════
%  POOL REQUIREMENTS (FULL AND INLAY — DRIVEN BY TYPE SELECTION)
% ══════════════════════════════════════════════════════════════════

\paragraph{Full and inlay pool requirements.}
The full and inlay pool allocations are determined once the type
is selected (Disjunction~1) and the shipment block is active
(Disjunction~3). They do not require separate leaders and are
stated as unconditional expressions using the aggregated forms
of Convention~\ref{conv:notation}:
%
\begin{align}
  TNCF_{cij}
  &= (1+\alpha^{f})\!\left(
       NS_{cij}
       + l_{ij}\cdot NF_{cij}
       + \overline{XF}_{cij}
         \bigl(ssf^{i,c}_{ij}+ssf^{j,c}_{ij}\bigr)
     \right)
  && \forall\,(i,j)\in E,\ c\in C
  \label{eq:pool-full}\\[4pt]
  TNCI_{cij}
  &= (1+\alpha^{f})\!\left(
       NS_{cij}
       + l_{ij}\cdot NI_{cij}
     \right)
  && \forall\,(i,j)\in E,\ c\in C
  \label{eq:pool-inlay}
\end{align}

% ══════════════════════════════════════════════════════════════════
%  RTI PURCHASE
% ══════════════════════════════════════════════════════════════════

\paragraph{RTI purchase.}
For each type the network purchases any shortfall between the
total pool requirement across all lanes and the existing stock;
a surplus remains idle:
%
\begin{align}
  \sum_{(i,j)\in E}\!\bigl(
    TNCF_{cij}+TNCE_{cij}+TNCI_{cij}
  \bigr)
  &\leq s^{\text{RTI},c} + P_{c}
  && \forall\, c\in C,\quad P_{c}\geq 0
  \label{eq:purchase}
\end{align}
