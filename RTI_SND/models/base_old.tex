\begin{longtable}{|p{0.19\textwidth}|p{0.8\textwidth}|}
\caption{Nomenclature of PLC-SCPTOP}
\label{tab:Nomenclature}\\
\hline
\multicolumn{2}{|l|}{\textbf{Indices}}\\ \hline
$V$ & Set of nodes of the supply chain, $v = 1, 2, \dots, |V|$ \\
$E$ & Set of edges, $E \subseteq V \times V$ \\
\hline
\multicolumn{2}{|l|}{\textbf{Parameters}}\\ \hline
$\mu_{ij}$ & Daily average of full RTIs going from $i$ to $j$\\
$ric_{ij}$ & Daily average of full boxes with return inlay from $i$ to $j$\\
$fr_{ij}$ & Daily average of fixed empty RTIs from $i$ to $j$ for type $c$ \\
$c_{ij}$ & Cost per volume unit from $i$ to $j$\\
$v^{f}$, $v^{e}$ & Volume of full ($v^{f}$) or empty ($v^{e}$) RTI\\
$\delta$ & Minimum volume for shipment. \\
$\omega$ & Maximum volume for a shipment.\\
$\gamma$
& Fractions of  RTIs that are out for maintenance (unavailable for use) per period. \\
\hline
\multicolumn{2}{|l|}{\textbf{Decision Variables}}\\ \hline
\multicolumn{2}{|l|}{Binary Variables}\\ \hline
$X_{ij}$ & Binary variables indicating if edge $(i,j)$ is used for any kind of transportation. \\
$XE_{ij}$, $XF_{ij}$, $XI_{ij}$ & Binary variables indicating if edge $(i,j) \in E$ transports empties RTIs ($XE_{ij}$), transports full RTIs ($XF_{ij}$), full RTIs with inlay ($XI_{ij}$)\\
$XInc_{i}$, $XOut_{i}$ & Binary variables indicating if node $i\in V$ has incoming or outcoming RTIs\\
\hline
\multicolumn{2}{|l|}{Number of Containers Variables}\\ \hline
$NE_{ij}$, $NF_{ij}$, $NI_{ij}$ & Number of emtpies, Number of full RTIs ($NF_{ij}$) and full with inlay RTIs ($NI_{ij}$) of type $c$ sent from $i$ to $j$, transport mode $m$, queue time $\tau_q$ \\
\hline
\multicolumn{2}{|l|}{Volume per day Variables}\\ \hline
$VE_{ij}$, $VF_{ij}$, $VI_{ij}$, $VR_{ij}$ & Volume of RTIs as above for type $c$ \\
\hline
\multicolumn{2}{|l|}{Volume per shipment Variables}\\ \hline
$VSE_{ij}$, $VSF_{ij}$, $VSI_{ij}$ & Volume of full RTIs ($VSF_{ij}$) and full with inlay RTIs ($VSI_{ij}$) of type $c$ sent from $i$ to $j$ on route $k$, transport mode $m$, queue time $\tau_q$ \\
\hline
\multicolumn{2}{|l|}{Total Number of Container Variables}\\ \hline
$TNCE_{ij}$, $TNCF_{ij}$ & Total number of returned and full containers for route $(i,j)$ using RTI type $c$, transport mode $m$, and queue time $\tau_q$ \\
$TNCF_{ij}$ & Total number of full containers for route $(i,j)$ using RTI type $c$, route $k$, transport mode $m$, and queue time $\tau_q$ \\
\hline
\multicolumn{2}{|l|}{Cost per day Variables}\\ \hline
$CE_{ij}$, $CI_{ij}$,  $CF_{ij}$ & Transport cost as above for RTI type $c$ \\
$CF_{ij}$ & Transport cost of full RTIs of type $c$ sent from $i$ to $j$ on route $k$, transport mode $m$, queue time $\tau_q$ \\ \hline
\multicolumn{2}{|l|}{Cost per shipment Variables}\\ \hline
$CSR_{ij}$, $CSF_{ij}$ & Transport cost as above for RTI type $c$ \\
$CSF_{ij}$ & Transport cost of full RTIs of type $c$ sent from $i$ to $j$ on route $k$, transport mode $m$, queue time $\tau_q$ \\ \hline
\end{longtable}

\begin{align}
    \sum_{j \in V} (NE_{cij} + NF_{cij} + NI_{cij}) 
    &= \sum_{j \in V} (NE_{cji} + NF_{cji} + NI_{cji}) 
    && \forall i \in V
    \label{eq:node_flow}
\end{align}

\begin{align}
    3 X_{ij} &\geq XE_{ij} + XF_{ij} + XI_{ij} 
    && \ i \in V,\ j \in V 
\end{align}

\begin{align}
  NF_{ij} &= \mu_{ij} && \forall \ i \in V,\ j \in V  \\
  NI_{ij} &= ric_{ij}\cdot\mu_{ij} && \forall \ i \in V,\ j \in V 
\end{align}


\begin{align}
    M \cdot XF_{ij} &\geq NF_{ij} &&  \forall \ i \in V,\ j \in V \\ 
    M \cdot XI_{ij} &\geq NI_{ij} &&  \forall \ i \in V,\ j \in V \\ 
    M \cdot XE_{ij} &\geq NE_{ij} &&  \forall \ i \in V,\ j \in V \\ 
\end{align}

\begin{align}
    NF_{ij} &\leq XF_{ij} \cdot M 
    && \forall c, i, j, m, q \\
    NE_{ij} &\leq XE_{ij} \cdot M 
    && \forall c, i, j, m, q \\
    NE_{ij} &\geq XE_{ij} \cdot fr_{cijqm} 
    && \forall c, i, j, m, q \\
    NI_{ij} &\leq XI_{ij} \cdot M 
    && \forall c, i, j, m, q \\
    NI_{ij} &= XF_{ij} \cdot ric_{cij} 
    && \forall c, i, j, m, q \\
    NF_{ij} &= \mu_{cij} \cdot XF_{ij} 
    && \forall c, i, j, m, q 
\end{align}

% Link number of containers per direct shipment
\begin{align}
    NSF_{ij} &= NF_{ij} \cdot \tau_q 
    && \forall c, i, j, m, q \\
    NSI_{ij} &= NI_{ij} \cdot \tau_q 
    && \forall c, i, j, m, q \\
    NSE_{ij} &= NE_{ij} \cdot \tau_q 
    && \forall c, i, j, m, q
\end{align}

\begin{align}
    VSF_{ij} &= VF_{ij} \cdot \tau_q 
    && \forall c, i, j, m, q \\
    VSI_{ij} &= VI_{ij} \cdot \tau_q 
    && \forall c, i, j, m, q \\
    VSE_{ij} &= VE_{ij} \cdot \tau_q 
    && \forall c, i, j, m, q
\end{align}

% Link number of containers and volume
\begin{align}
    VE_{ij} &= NE_{ij} \cdot v^e_c 
    && \forall c, i, j, m, q \\
    VI_{ij} &= NI_{ij} \cdot v^f_c 
    && \forall c, i, j, m, q \\
    VF_{ij} &= NF_{ij} \cdot v^f_c 
    && \forall c, i, j, m, q
\end{align}

% Link volume of shipment mode and q day
\begin{align}
    \sum_{c} (VSE_{ij} + VSF_{ij} + VSI_{ij}) 
    &\geq \omega^m_q \cdot X_{ijmq} 
    && \forall i, j, m, q \\
    \sum_{c} (VSE_{ij} + VSF_{ij} + VSI_{ij}) 
    &\leq \delta^m_q \cdot X_{ijmq} 
    && \forall i, j, m, q
\end{align}

\begin{align}
    TNCF_{ij} &= (1+\alpha^f)\cdot(NSF_{ij} + l_{ij} \cdot NF_{ij} + XF_{ij} \cdot (ssf^i_{cij} + ssf^j_{cij})) 
    && \forall c, i, j, m, q \\
    TNCE_{ij} &= (1+\alpha^e)\cdot(NSE_{ij} + l_{ij} \cdot NE_{ij} + XE_{ij} \cdot (sse^i_{cij} + sse^j_{cij})) 
    && \forall c, i, j, m, q \\
    TNCI_{ij} &=(1+\alpha^f)\cdot( NSI_{ij} + l_{ij} \cdot NI_{ij} )
    && \forall c, i, j, m, q
\end{align}
